\section{Bayesian Optimization}
\textbf{MAB}: We are given a set of $k$ actions, and want to maximize reward.
\begin{framed}
    The \textbf{Regret} for a time horizon $T$ associated with choices $\{\vx_t\}_{t=1}^T$ is defined as: $R_T \defeq \sum_{t=1}^T \underbrace{\left({\max_\vx \opt{f}(\vx) - \opt{f}(\vx_t)}\right)}_{\text{\textit{instantaneous regret}}}$.
\end{framed}
Goal: \textbf{sublinear regret}: $\lim_{T\to\infty} \frac{R_T}{T} = 0$.
\textbf{Acquisition function} used to greedily pick the next point to sample based on the current model. \\
\textbf{Well-calibrated confidence intervals}: with probability $\ge 1-\delta$:
$\forall x\in\spX: f^\star(x)\in\mathcal C_t(x)=[\mu_t(x) - \beta_t \sigma_t(x), \mu_t(x) + \beta_t \sigma_t(x)]$.\\
\textbf{UCB}: $\vx_{t+1} \defeq \argmax_{\vx \in \spX} \mu_{t}(\vx) + \beta_{t+1} \sigma_{t}(\vx)$
Choosing $\beta_t(\delta)\in \BigO{\sqrt{\log(\nicefrac{|\spX|t}{\delta})}}$ appropriately we get: $R_T = \BigO{\sqrt{T \gamma_T}}$, 
where $\gamma_T \defeq \max_{\substack{\sS \subseteq \spX \\ \card{\sS} = T}} \I{\vf_\sS}{\vy_\sS} = \max_{\substack{\sS \subseteq \spX \\ |\sS| = T}} \frac{1}{2} \log\det{\mI + \sigman^{-2} \mK_{\sS\sS}}$, is the maximum information gain after $T$ rounds.
\begin{framed}
    \textbf{Information gain of some kernels}:\\
    Linear: $\gamma_T = \BigO{d \log T}$ \\
    Gaussian: $\gamma_T = \BigO{(\log T)^{d+1}}$ \\
\end{framed}
\textbf{Improvement}: $I_t(x)\defeq \max\{f(x)-\hat{f_t},0\}$\\
\textbf{PI}: $x_{t+1} = \arg\max_{x \in \spX} \mathbb P(I_t(x) > 0)$\\
\textbf{EI}: $x_{t+1} = \arg\max_{x \in \spX} \mathbb E[I_t(x)]$\\
\textbf{Thompson Sampling}: sample $\Tilde{f}_{t+1} \sim p(\cdot \mid \vx_{1:t}, y_{1:t})$
and select $\vx_{t+1} \defeq \argmax_{\vx \in \spX} \Tilde{f}_{t+1}(\vx)$.\\
\textbf{Information ratio}: $\Psi_t(x)\defeq \frac{\Delta(x)^2}{I_t(x)}$, with $\Delta(x)\defeq \max_{x'}f^\star(x')-f^\star(x)$\\
\textbf{IDS}: $x_{t+1} = \arg\min_{x \in \spX}\left\{ \hat\Psi_t(x)=\frac{\hat\Delta_t(x)^2}{I_t(x)} \right\}$ with $\hat\Delta_t(x)\defeq \max_{x'}u_t(x')-l_t(x)$\\
